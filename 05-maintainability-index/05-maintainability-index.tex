\plush{
\pptBanner{The Formula of Maintainability Index}
\begin{multicols}{2}
\pptPic{.95}{formula-3.png}
\par\columnbreak\par
\textit{aveVol} --- average Halstead Volume in a module\par
\textit{ave V(g')} --- average total cyclomatic complexity in a module\par
\textit{aveLOC} --- average lines of code in a module\par
\textit{perCM} --- average percent of comments in a module\par
\end{multicols}
}

\pitch{\pptQuote{rainer-niedermayr.jpg}{We are convinced that Maintainability Index is \emph{nonsense}. We think that it is not sensible to reduce the maintainability of a whole software system to one single indicator.}{Rainer Niedermayr, \textit{Why we don’t use the Software Maintainability Index}, 2016, \href{https://www.cqse.eu/de/news/blog/maintainability-index/}{link}}}

\plush{\pptThought{``The Maintainability Index does not provide information about the impact on development activities. A value of 57 does not express which maintainability aspects are affected by a bad value.'' --- Rainer Niedermayr}}

\pitch{\pptQuote{arie-van-deursen.jpg}{If you are a researcher, \emph{think twice} before using the maintainability index in your experiments. Make sure you study and fully understand the original papers published about it.}{Arie van Deursen, \textit{Think Twice Before Using the ``Maintainability Index''}, 2014, \href{https://avandeursen.com/2014/08/29/think-twice-before-using-the-maintainability-index/}{link}}}

\plush{\pptThought{``Tool smiths and vendors used the exact same formula and coefficients as the 1994 experiments, without any recalibration.'' --- Arie van Deursen}}

\pitch{\pptQuote{tim-gilboy.jpg}{If we're going to use the Maintainability Index we should use it to measure \emph{relative} maintainability within our project rather than use it as an \emph{absolute} metric.}{Tim Gilboy, \textit{Maintainability Index - What is it and where does it fall short?}, 2022, \href{https://sourcery.ai/blog/maintainability-index/}{link}}}

\plush{\pptThought{\raggedright``Extending the length can significantly decrease Maintainability Index, even if all of the changes cause the code to be clearer and more understandable.'' --- Tim Gilboy}}

\pitch{\pptQuote{hericko.png}{When comparing maintainability measurements from several Index variants, the perception of maintainability could be impacted by the choice of the Index variant used.}{Tja\u{s}a Heri\u{c}ko et al., \textit{Exploring Maintainability Index Variants for Software Maintainability Measurement in Object-Oriented Systems}, Applied Sciences, 2023}}

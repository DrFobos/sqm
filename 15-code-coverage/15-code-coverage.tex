\pptBanner{Example, Part I}
\begin{pptWide}{2}
Live Code:\par
{\small\begin{ffcode*}{highlightlines={2-4}}
int fibonacci(int n) {
  if (n <= 2) {
    return 1;
  }
  return fibonacci(n-1)
    + fibonacci(n-2);
}
\end{ffcode*}
}
\par\columnbreak\par
Test Code:\par
{\small\begin{ffcode*}{}
assert fibonacci(0) == 1;
assert fibonacci(1) == 1;
assert fibonacci(2) == 1;
\end{ffcode*}
}

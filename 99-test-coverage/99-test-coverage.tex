\pptBanner{Motivating Example}
\begin{multicols}{2}
Live code:\par
{\small\begin{ffcode}
class Fibonacci
  int eval(int n)
    if (n <= 2)
      return 1;
    return eval(n - 1)
      + eval(n - 2);
\end{ffcode}
}
\par\columnbreak\par
Unit test:\par
{\small\begin{ffcode}
class FibonacciTest
  @Test
  void worksJustFine()
    var f = new Fibonacci();
    assert(f.eval(0) == 1);
    assert(f.eval(1) == 1);
    assert(f.eval(2) == 1);
\end{ffcode}
}
\end{multicols}

\pptBanner{Which One Is Better?}
\begin{multicols}{2}
{\small\begin{ffcode}
int f(int n)
{
  if (n == 1 || n < 2)
    return 1;
  int r = f (n-1);
  int r2 = f(n - 2);
  return r +r2;
  }
\end{ffcode}
}
\par\columnbreak\par
{\small\begin{ffcode}
int fibonacci(int n) {
  if (n <= 2) {
    return 1;
  }
  return fibonacci(n - 1)
    + fibonacci(n - 2);
}
\end{ffcode}
}
\end{multicols}
